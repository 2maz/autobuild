        def installstamp
            std_stamp = super
            return std_stamp if File.file?(std_stamp)

            pcfile = File.join(pkgconfig.prefix, "lib", "pkgconfig", "#{name}.pc")
            if !File.file?(pcfile)
                raise "cannot find the .pc file for #{name}, tried #{pcfile}"
            end
            pcfile
        end
