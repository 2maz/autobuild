Autobuild::FakeOrogenEnvironment

This discards everything but the calls to import_types_from, using_task_library and using_toolkit. This is used to automatically discover the dependencies without resorting to an actual build

        def self.orocos_target
            user_target = ENV['OROCOS_TARGET']
            if @orocos_target
                @orocos_target.dup
            elsif user_target && !user_target.empty?
                user_target
            else
                'gnulinux'
            end
