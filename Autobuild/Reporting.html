Autobuild::Reporting

 The reporting module provides the framework

to run commands in autobuild and report errors to the user

It does not use a logging framework like Log4r, but it should ;-)

        def self.each_log
            Find.find(Autobuild.logdir) do |path|
                if File.file?(path) && path =~ /\.log$/
                    yield(path)
                end
            end
        end

        def self.report
            begin
                yield
            rescue Autobuild::Exception => e
                error(e)
                exit(1) if e.fatal?
            end
        end
