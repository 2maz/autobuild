        def initialize(*args)
            @exclude = []
            super
            exclude << Regexp.new("^#{Regexp.quote(installstamp)}")

            Autobuild.source_tree(srcdir) do |pkg|
                pkg.exclude.concat exclude
                exclude.freeze
            end

            file installstamp => srcdir do 
                Autobuild.touch_stamp installstamp
            end
        end
