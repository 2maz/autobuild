    def initialize(name)
        if !system("pkg-config --exists #{name}")
            raise NotFound.new(name)
        end
        
        @name    = name
        @version = `pkg-config --modversion #{name}`.chomp.strip
        @actions = Hash.new
        @variables = Hash.new
    end
